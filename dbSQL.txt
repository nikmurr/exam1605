-- Создание таблицы user_roles
CREATE TABLE user_roles (
    id SERIAL PRIMARY KEY,
    role_name VARCHAR(64) NOT NULL
);

-- Создание таблицы users
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    role_id INTEGER NOT NULL REFERENCES user_roles(id),
    login VARCHAR(32) UNIQUE NOT NULL,
    password VARCHAR(64) NOT NULL
);

-- Создание таблицы education_types
CREATE TABLE education_types (
    id SERIAL PRIMARY KEY,
    type_name VARCHAR(20) NOT NULL
);

-- Создание таблицы specialties
CREATE TABLE specialties (
    id SERIAL PRIMARY KEY,
    education_type_id INTEGER NOT NULL REFERENCES education_types(id),
    specialty_name VARCHAR(256) NOT NULL
);

-- Создание таблицы applicants
CREATE TABLE applicants (
    id SERIAL PRIMARY KEY,
    user_id INTEGER UNIQUE NOT NULL REFERENCES users(id),
    education_type_id INTEGER NOT NULL REFERENCES education_types(id),
    full_name VARCHAR(128) NOT NULL,
    passport_number VARCHAR(32) UNIQUE NOT NULL,
    SNILS VARCHAR(32) UNIQUE NOT NULL,
    email VARCHAR(64) NOT NULL,
    phone VARCHAR(32) NOT NULL,
    parent_full_name VARCHAR(128) NOT NULL,
    institution_attended VARCHAR(64) NOT NULL,
    diploma_score FLOAT,
    certificate_score FLOAT,
    ege_score FLOAT,
    specialty_id INTEGER NOT NULL REFERENCES specialties(id),
    education_document_scan VARCHAR(256)
);
